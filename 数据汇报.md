### 1. 保存数据汇报模板
#### 描述
保存数据汇报模板
#### 请求
POST /office/dataReport/template
新增示例
```json
    {
        "departmentId": "402880086950ab4c016950b1a4380001",
        "name": "Java模板",
        "period": "DAY",
        "days": [0,1,3], 
        "startDate": 0,
        "endDate": 0,
        "startTime": 8,
        "endTime": 24,
        "remindContext": "提醒内容",
        "remindNotification": true,
        "remindTime": 60,
        "items": [
            {
                "name": "修理bug",
                "target": 5,
                "unit": "个",
                "ratingSettings": [
                 {
                    "value": 10,
                    "score": 100,
                    "operator": "PLUS"
                 }
                ]
            }, {
              "name": "测试bug",
               "target": 10,
               "unit": "个"
            }
         ]
      
    }
```

```
    {
        "departmentId": "部门ID",

        "period": "提交周期"，类型：DAY("天")WEEK("周")MONTH("月");

        "endDate": 0,结束日期
        "startTime": 8,开始时间 8 = 8：00
        "endTime": 24,结束时间 24 = 24：00
        "remindContext": "提醒内容",
        "remindNotification": true,是否开始提醒
        "remindTime": 60,截止时间前一小时=60，截止时间前两小时=120
        "items": [ //数据项目
            {
                "name": "名称",
                "target": 达标值,
                "unit": "单位",
                "ratingSettings": [ 考核设置
                 {
                    "value": 考核数值,
                    "score": 分数,
                    "operator": "运算符" PLUS：加，SUBTRACT：减
                 }
            }, {
                "name": "名称",
                "target": 达标值,
                "unit": "单位"
            }
         ]
      
    }
```

200
